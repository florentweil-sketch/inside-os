Pseudo-code :

function computeStrategicScore(axes) {
  const {
    financial_impact,
    systemic_impact,
    urgency,
    risk_if_ignored
  } = axes;

  return (
    financial_impact * 3 +
    systemic_impact * 2 +
    urgency * 2 +
    risk_if_ignored * 3
  );
}

function deriveStrategicLevel(score) {
  if (score <= 10) return "Bruit";
  if (score <= 20) return "Faible";
  if (score <= 30) return "Important";
  if (score <= 40) return "Majeur";
  return "Critique";
}

Tu peux changer les poids, mais pas le principe : discret + explicable.

	•	levier: 0..5
	•	risque: 0..5
	•	effort: 0..5
	•	recurrence: 0..5
	•	confidence: low|medium|high

function confidenceBonus(c) {

  if (c === "medium") return 0;
  return -10;
}

function score({impact, urgence, levier, risque, effort, recurrence, confidence}) {
  const raw =
    impact*12 + urgence*10 + levier*10 + risque*8 - effort*6 + recurrence*6 + confidenceBonus(confidence);
  return Math.max(0, Math.min(100, raw));
}
