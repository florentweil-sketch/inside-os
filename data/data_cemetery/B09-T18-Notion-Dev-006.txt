os/pipeline.mjs

#!/usr/bin/env node

import { spawn } from "node:child_process";
import process from "node:process";
import path from "node:path";
import { fileURLToPath } from "node:url";

const __filename = fileURLToPath(import.meta.url);
const __dirname = path.dirname(__filename);
const ROOT_DIR = path.resolve(__dirname, "..");

const STEPS = [
  {
    name: "ingest",
    script: path.join(ROOT_DIR, "os", "ingest", "ingest-thread-dump.mjs"),
  },
  {
    name: "extract",
    script: path.join(ROOT_DIR, "os", "extract", "extract-thread-dump.mjs"),
  },
  {
    name: "inject",
    script: path.join(ROOT_DIR, "os", "inject", "inject-decisions-lessons.mjs"),
  },
];

function parseArgs(argv) {
  const parsed = {
    only: undefined,
    force: false,
    passthrough: [],
  };

  for (let i = 0; i < argv.length; i += 1) {
    const arg = argv[i];

    if (arg === "--only") {
      const value = argv[i + 1];
      if (!value || value.startsWith("--")) {
        throw new Error("Missing value for --only");
      }
      parsed.only = value;
      i += 1;
      continue;
    }

    if (arg.startsWith("--only=")) {
      parsed.only = arg.slice("--only=".length);
      if (!parsed.only) {
        throw new Error("Missing value for --only");
      }
      continue;
    }

    if (arg === "--force") {
      parsed.force = true;
      continue;
    }

    parsed.passthrough.push(arg);
  }

  return parsed;
}

function buildStepArgs(parsed) {
  const args = [];

  if (parsed.only) {
    args.push("--only", parsed.only);
  }

  if (parsed.force) {
    args.push("--force");
  }

  if (parsed.passthrough.length > 0) {
    args.push(...parsed.passthrough);
  }

  return args;
}

function formatCommand(command, args) {
  return [command, ...args]
    .map((part) => {
      if (/[\s"]/u.test(part)) {
        return JSON.stringify(part);
      }
      return part;
    })
    .join(" ");
}

function runNodeScript(step, forwardedArgs) {
  return new Promise((resolve, reject) => {
    const command = process.execPath;
    const args = [step.script, ...forwardedArgs];

    console.log(`\n[os:pipeline] step=${step.name}`);
    console.log(`[os:pipeline] exec> ${formatCommand(command, args)}`);

    const child = spawn(command, args, {
      cwd: ROOT_DIR,
      stdio: "inherit",
      env: process.env,
    });

    child.on("error", (error) => {
      reject(
        new Error(
          `[os:pipeline] Failed to start step "${step.name}": ${error.message}`
        )
      );
    });

    child.on("exit", (code, signal) => {
      if (signal) {
        reject(
          new Error(
            `[os:pipeline] Step "${step.name}" terminated by signal ${signal}`
          )
        );
        return;
      }

      if (code !== 0) {
        reject(
          new Error(
            `[os:pipeline] Step "${step.name}" failed with exit code ${code}`
          )
        );
        return;
      }

      resolve();
    });
  });
}

async function main() {
  const rawArgs = process.argv.slice(2);
  const parsed = parseArgs(rawArgs);
  const forwardedArgs = buildStepArgs(parsed);

  console.log("[os:pipeline] starting");
  console.log(
    `[os:pipeline] forwarded args: ${
      forwardedArgs.length > 0 ? forwardedArgs.join(" ") : "(none)"
    }`
  );

  for (const step of STEPS) {
    await runNodeScript(step, forwardedArgs);
  }

  console.log("\n[os:pipeline] done");
}

main().catch((error) => {
  console.error(`\n[os:pipeline] ERROR: ${error.message}`);
  process.exit(1);
});

Pourquoi cette version est la bonne
Parce qu'elle fait le job sans te vendre du rêve :
	-	npm run os:pipeline -- --only B00-T00 devient une vraie exécution Node avec vrais arguments transmis aux 3 étapes.
	-	npm run os:pipeline -- --only B00-T00 --force transmet aussi --force partout.
	-	npm run os:pipeline exécute le pipeline sans flags.
Et surtout : le contrat devient lisible. Tu peux l'étendre plus tard sans te retrouver dans un marécage shell/npm.

1. Code de base - validate-schema.mjs

import 'dotenv/config'
import { Client } from '@notionhq/client'

const notion = new Client({ auth: process.env.NOTION_TOKEN })

const DBS = {
  THREAD_DUMP: process.env.THREAD_DUMP_DB_ID,
  DECISIONS: process.env.DECISIONS_DB_ID,
  LESSONS: process.env.LESSONS_DB_ID
}

const EXPECTED = {
  THREAD_DUMP: {
    id_dump: 'title',
    raw_text: 'rich_text',
    created_at: 'date'
  },

  DECISIONS: {
    uid: 'rich_text',
    decision: 'rich_text',
    status: 'select',
    source_thread: 'rich_text'
  },

  LESSONS: {
    uid: 'rich_text',
    lesson: 'rich_text',
    source_thread: 'rich_text'
  }
}

function fail(msg) {
  console.error(`[validate-schema] ERROR: ${msg}`)
  process.exit(1)
}

async function validateDatabase(name, dbId) {

  if (!dbId) {
    fail(`${name} DB id missing in env`)
  }

  const db = await notion.databases.retrieve({ database_id: dbId })

  const props = db.properties

  const expected = EXPECTED[name]

  console.log(`[validate-schema] Checking ${name}`)

  for (const key of Object.keys(expected)) {

    if (!props[key]) {
      fail(`${name}: missing property "${key}"`)
    }

    const type = props[key].type
    const expectedType = expected[key]

    if (type !== expectedType) {
      fail(`${name}: property "${key}" expected type "${expectedType}" but got "${type}"`)
    }
  }

  console.log(`[validate-schema] ${name} OK`)
}

async function main() {

  console.log('[validate-schema] starting')

  await validateDatabase('THREAD_DUMP', DBS.THREAD_DUMP)
  await validateDatabase('DECISIONS', DBS.DECISIONS)
