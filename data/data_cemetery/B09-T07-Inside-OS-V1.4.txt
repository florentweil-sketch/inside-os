1) Seed "propre" : la règle d'or (sinon tu vas te faire poignarder par les doublons)
Chaque base doit avoir une clé canonique stable (un "natural key"), indépendante de Notion.
Je te conseille ce format :
	-	entities : entity_key = slug stable (ex: inside-sas, atelier-colombe)
	-	digital_assets : asset_key = entity_key + ":" + type + ":" + provider + ":" + handle_or_domain
	-	ex: inside-sas:web:domain:inside.fr
	-	ex: inside-sas:social:instagram:inside
	-	projects_* : project_key = entity_key + ":" + year + ":" + shortname
	-	ex: inside-sas:2026:chantier-roussel-grange
	-	decisions_structural : decision_key = scope + ":" + yyyy-mm + ":" + shortslug
	-	lessons_learnings : lesson_key = project_key + ":" + yyyy-mm-dd + ":" + shortslug
 Ensuite, toutes les relations (entité <-> actif <-> projet) se font via ces clés au moment de l'ingestion.
Action immédiate : ajoute dans chaque base une propriété texte *_key + une propriété source_hash (sha256 du payload source normalisé). Ça te donne : idempotence + dédup + audit.

2) Pipeline CSV -> Notion : un vrai "upsert" (pas un import de touriste)
Objectif : tu balances un CSV, et le système fait :
	1	normalisation (trim, lower, formats, dates)
	2	mapping vers ton schéma
	3	détection doublon (par *_key)
	4	create si absent, update si présent (optionnel : update seulement si source_hash change)
	5	résolution des relations (lookup des pages liées via leurs keys)
Stratégie anti-doublon (robuste et simple)
	-	Tu fais un query Notion "Database" filtré sur *_key == value.
	-	Si trouvé : update
	-	Sinon : create
Option bonus (ultra propre) : tu maintiens un cache local cache/notion-id-map.json (key -> page_id) régénéré au démarrage par un scan Notion (1 fois), pour accélérer.
Squelette Node (minimal, efficace)

Pseudo-code upsert :

async function upsertByKey({ dbId, keyProp, keyValue, properties }) {
  const existing = await notion.databases.query({
    database_id: dbId,
    filter: { property: keyProp, rich_text: { equals: keyValue } },
    page_size: 1,
  });

  if (existing.results.length) {
    const pageId = existing.results[0].id;
    return notion.pages.update({ page_id: pageId, properties });
  }
  return notion.pages.create({ parent: { database_id: dbId }, properties });
}

require("dotenv").config();
const { Client } = require("@notionhq/client");
const notion = new Client({ auth: process.env.NOTION_TOKEN });

function rt(text) {
  return [{ type: "text", text: { content: String(text ?? "") } }];
}

async function findByRichTextEquals({ database_id, property, value }) {
  const res = await notion.databases.query({
    database_id,
    page_size: 1,
    filter: {
      property,
      rich_text: { equals: String(value ?? "") },
    },
  });
  return res.results?.[0] || null;
}

async function upsertByRichTextKey({ database_id, key_property, key_value, createProps, updateProps }) {
  const existing = await findByRichTextEquals({
    database_id,
    property: key_property,
    value: key_value,
  });

  if (!existing) {
    const created = await notion.pages.create({
      parent: { database_id },
      properties: createProps,
    });
    return { action: "created", page: created };
  }

  const updated = await notion.pages.update({
    page_id: existing.id,
    properties: updateProps,
  });
  return { action: "updated", page: updated };
}

module.exports = { notion, rt, upsertByRichTextKey };

require("dotenv").config();
const { Client } = require("@notionhq/client");

const notion = new Client({ auth: process.env.NOTION_TOKEN });

function rt(text) {
  return [{ type: "text", text: { content: String(text ?? "") } }];
}

async function findByRichTextEquals({ database_id, property, value }) {
  const res = await notion.databases.query({
    database_id,
    page_size: 1,
    filter: {
      property,
      rich_text: { equals: String(value ?? "") },
    },
  });
  return res.results?.[0] || null;
}

async function upsertByRichTextKey({ database_id, key_property, key_value, createProps, updateProps }) {
  const existing = await findByRichTextEquals({
    database_id,
    property: key_property,
    value: key_value,
  });

  if (!existing) {
    const created = await notion.pages.create({
      parent: { database_id },
      properties: createProps,
    });
    return { action: "created", page: created };
  }

  const updated = await notion.pages.update({
    page_id: existing.id,
    properties: updateProps,
  });
  return { action: "updated", page: updated };
}

async function findByRichTextEquals({ database_id, property, value }) {
  const filter = {
    property,
    rich_text: { equals: String(value ?? "") },
  };

  const json = await dbQueryHTTP({ database_id, filter });
  return json.results?.[0] || null;
}

async function upsertByRichTextKey({ database_id, key_property, key_value, createProps, updateProps }) {
  const existing = await findByRichTextEquals({
    database_id,
    property: key_property,
    value: key_value,
  });

  if (!existing) {
    const created = await notion.pages.create({
      parent: { database_id },
      properties: createProps,
    });
    return { action: "created", page: created };
  }

  const updated = await notion.pages.update({
    page_id: existing.id,
    properties: updateProps,
  });
  return { action: "updated", page: updated };
}
