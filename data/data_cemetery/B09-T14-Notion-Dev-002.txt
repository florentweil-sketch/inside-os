import "dotenv/config";
import fs from "fs";
import path from "path";
import { Client } from "@notionhq/client";

// ---- CLI args (minimal) ----
const argv = process.argv.slice(2);
const getArg = (name, def = null) => {
  const i = argv.indexOf(name);
  if (i === -1) return def;
  const v = argv[i + 1];
  if (!v || v.startsWith("--")) return true;
  return v;
};

const OUT_DIR = getArg("--out", "out/audit");
const SAMPLE_PAGES = Number(getArg("--sample", "5"));        // pages per DB
const MAX_DBS = Number(getArg("--max-dbs", "9999"));         // safety cap

const NOTION_TOKEN = process.env.NOTION_TOKEN;
if (!NOTION_TOKEN) {
  console.error("[audit] ERROR: NOTION_TOKEN missing in env");
  process.exit(1);
}

const notion = new Client({ auth: NOTION_TOKEN });

const mkdirp = (p) => fs.mkdirSync(p, { recursive: true });
const writeJson = (file, obj) => fs.writeFileSync(file, JSON.stringify(obj, null, 2), "utf8");

mkdirp(OUT_DIR);

// ---- Small utils ----
function safeName(s) {
  return String(s || "db")
    .replaceAll("/", "_")
    .replaceAll("\\", "_")
    .replaceAll(":", "_")
    .replaceAll("*", "_")
    .replaceAll("?", "_")
    .replaceAll('"', "_")
    .replaceAll("<", "_")
    .replaceAll(">", "_")
    .replaceAll("|", "_")
    .trim()
    .slice(0, 80);
}

function getPlainText(arr) {
  return Array.isArray(arr) ? arr.map(x => x?.plain_text || "").join("").trim() : "";
}

function getContainerTitle(obj) {
  // Legacy database object
  if (obj?.title) {
    const t = getPlainText(obj.title);
    if (t) return t;
  }

  // Newer data_source object
  if (obj?.name) return String(obj.name).trim();

  // Sometimes exposed through parent database metadata
  if (obj?.parent?.database_id) return `(data_source ${obj.parent.database_id})`;

  return "(untitled)";
}

// ---- Notion helpers ----
async function searchAllContainers() {
  const all = [];
  let start_cursor = undefined;

  // 1) Modern search target
  try {
    while (true) {
      const res = await notion.search({
        filter: { property: "object", value: "data_source" },
        start_cursor,
        page_size: 100,
      });
      all.push(...(res.results || []));
      if (!res.has_more) break;
      start_cursor = res.next_cursor;
    }

    console.log(`[audit] search mode=data_source, found=${all.length}`);
    return { mode: "data_source", items: all };
  } catch (e) {
    console.log("[audit] search data_source failed, trying legacy database mode...");
    console.log(`[audit] search data_source error: ${e?.message || String(e)}`);
  }

  // 2) Fallback legacy mode
  start_cursor = undefined;
  while (true) {
    const res = await notion.search({
      filter: { property: "object", value: "database" },
      start_cursor,
      page_size: 100,
    });
    all.push(...(res.results || []));
    if (!res.has_more) break;
    start_cursor = res.next_cursor;
  }

  console.log(`[audit] search mode=database, found=${all.length}`);
  return { mode: "database", items: all };
}

async function querySamplePages(database_id, page_size) {
  // Keep legacy query path for now.
  // If your workspace/API has fully switched, this is the next point that may require patching.
  const res = await notion.databases.query({
    database_id,
    page_size,
    sorts: [{ timestamp: "last_edited_time", direction: "descending" }],
  });
  return res.results || [];
}

async function retrieveSchema(database_id) {
  return notion.databases.retrieve({ database_id });
}

// ---- Main ----
(async () => {
  console.log(`[audit] OUT_DIR=${OUT_DIR}`);
  console.log(`[audit] SAMPLE_PAGES=${SAMPLE_PAGES}`);

  const search = await searchAllContainers();
  const dbs = search.items;

  console.log(`[audit] accessible containers: ${dbs.length}`);

  // Write index
  const index = dbs.map(db => ({
    id: db.id,
    object: db.object,
    title: getContainerTitle(db),
    url: db.url,
    created_time: db.created_time,
    last_edited_time: db.last_edited_time,
    parent: db.parent || null,
  }));
  writeJson(path.join(OUT_DIR, "index.containers.json"), index);

  // Per DB export (schema + sample pages)
  let count = 0;
  for (const db of dbs) {
    count += 1;
    if (count > MAX_DBS) break;

    const title = getContainerTitle(db);
    const slug = `${String(count).padStart(3, "0")}_${safeName(title)}_${String(db.id).replaceAll("-", "")}`;
    const dir = path.join(OUT_DIR, slug);
    mkdirp(dir);

    // full db schema
    let fullDb;
    try {
      fullDb = await retrieveSchema(db.id);
      writeJson(path.join(dir, "database.schema.json"), fullDb);
    } catch (e) {
      writeJson(path.join(dir, "database.schema_error.json"), {
        message: e?.message || String(e),
        name: e?.name,
        code: e?.code,
        object: db,
      });
      console.log(`[audit] ${title} -> schema ERROR (saved)`);
      continue;
    }

    // sample pages
    let pages = [];
    try {
      pages = await querySamplePages(db.id, SAMPLE_PAGES);
    } catch (e) {
      writeJson(path.join(dir, "database.query_error.json"), {
        message: e?.message || String(e),
        name: e?.name,
        code: e?.code,
      });
      console.log(`[audit] ${title} -> query ERROR (saved), schema OK`);
      continue;
    }

    writeJson(path.join(dir, "pages.sample.json"), pages);

    console.log(`[audit] OK: ${title} (schema + ${pages.length} pages)`);
  }

  console.log("[audit] DONE");
})().catch(err => {
  console.error("[audit] FATAL", err);
  process.exit(1);
});

  // Write index
  const index = dataSources.map(ds => ({

    object: ds.object,
    title: getDataSourceTitle(ds),
    url: ds.url,
    created_time: ds.created_time,
    last_edited_time: ds.last_edited_time,
    parent: ds.parent || null,
    parent_database_id: ds.parent?.database_id || null,
  }));
  writeJson(path.join(OUT_DIR, "index.data_sources.json"), index);

  // Per data source export (schema + sample pages)
  let count = 0;
  for (const ds of dataSources) {
    count += 1;
    if (count > MAX_DBS) break;

    const title = getDataSourceTitle(ds);
    const slug = `${String(count).padStart(3, "0")}_${safeName(title)}_${String(ds.id).replaceAll("-", "")}`;
    const dir = path.join(OUT_DIR, slug);
    mkdirp(dir);

    // full data source schema
    let fullDs;
    try {
      fullDs = await retrieveDataSource(ds.id);
      writeJson(path.join(dir, "data_source.schema.json"), fullDs);
    } catch (e) {
      writeJson(path.join(dir, "data_source.schema_error.json"), {
        message: e?.message || String(e),
        name: e?.name,
        code: e?.code,
        object: ds,
      });
      console.log(`[audit] ${title} -> schema ERROR (saved)`);
      continue;
    }

    // sample pages
    let pages = [];
    try {
      pages = await querySamplePages(ds.id, SAMPLE_PAGES);
    } catch (e) {
      writeJson(path.join(dir, "data_source.query_error.json"), {

        name: e?.name,
        code: e?.code,
      });
      console.log(`[audit] ${title} -> query ERROR (saved), schema OK`);
      continue;
    }

    writeJson(path.join(dir, "pages.sample.json"), pages);
